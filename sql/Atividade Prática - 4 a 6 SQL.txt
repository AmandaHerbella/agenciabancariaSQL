--Parte 1:

--1

SELECT UPPER(cliente_nome) AS nome_maiusculo
FROM cliente;

--2

SELECT INITCAP(cliente_nome) AS nome_formatado
FROM cliente;

--3

SELECT SUBSTR(cliente_nome, 1, 3) AS tres_primeiras_letras
FROM cliente;

--4

SELECT cliente_nome, LENGTH(cliente_nome) AS num_caracteres
FROM cliente;

--5

SELECT conta_numero, ROUND(saldo) AS saldo_arredondado
FROM conta;

--6

SELECT conta_numero, TRUNC(saldo) AS saldo_truncado
FROM conta;

--7

SELECT conta_numero, MOD(saldo, 1000) AS resto_divisao
FROM conta;

--8

SELECT SYSDATE AS data_atual
FROM dual;

--9

SELECT SYSDATE + 30 AS data_vencimento_simulada
FROM dual;

--10

SELECT conta_numero, SYSDATE - data_abertura AS dias_abertos
FROM conta;

--Parte 2

--11

SELECT conta_numero, TO_CHAR(saldo, 'L999G999D99') AS saldo_formatado
FROM conta;

--12

SELECT conta_numero, TO_CHAR(data_abertura, 'DD/MM/YYYY') AS data_formatada
FROM conta;

--13

SELECT conta_numero, NVL(saldo, 0) AS saldo_tratado
FROM conta;

--14

SELECT cliente_nome, NVL(cidade, 'Sem cidade') AS cidade_tratada
FROM cliente;

--15

SELECT cliente_nome,
       CASE 
           WHEN cidade = 'Niterói' THEN 'Região Metropolitana'
           WHEN cidade = 'Resende' THEN 'Interior'
           ELSE 'Outra Região'
       END AS grupo_regional
FROM cliente;

--Parte 3

--16

SELECT c.cliente_nome, a.conta_numero, a.saldo
FROM cliente c
JOIN conta a ON c.cliente_cod = a.cliente_cliente_cod;

--17

SELECT cl.cliente_nome, ag.agencia_nome
FROM cliente cl
JOIN conta co ON cl.cliente_cod = co.cliente_cliente_cod
JOIN agencia ag ON co.agencia_agencia_cod = ag.agencia_cod;


--18

SELECT ag.agencia_nome, cl.cliente_nome
FROM agencia ag
LEFT JOIN conta co ON ag.agencia_cod = co.agencia_agencia_cod
LEFT JOIN cliente cl ON co.cliente_cliente_cod = cl.cliente_cod;
